\usepackage{geometry}
\geometry{

\reversemarginpar         % marginnotes à gauche (oneside)

% ── Couleurs ────────────────────────────────────────────────
\usepackage{xcolor}
\definecolor{cifasBleu}{RGB}{0,82,155}
\definecolor{cifasGris}{RGB}{90,90,90}

% ── Marge : jour rotatif + bloc encadré ─────────────────────
\usepackage{marginnote}
\usepackage{rotating}

% Jour → texte rotatif dans la marge
\newcommand{\cifasDayNote}[1]{%
  \marginnote{%
    \rotatebox{90}{%
      \hspace{0pt}\footnotesize\sffamily\bfseries\MakeUppercase{#1}\hspace{0pt}%
    }%
  }[2pt]%
}

% Bloc → boîte dans la marge
\newcommand{\cifasBlocNote}[1]{%
  \marginnote{%
    {\setlength{\fboxsep}{4pt}\fbox{\footnotesize\sffamily\bfseries\MakeUppercase{#1}}}%
  }[2pt]%
}

% ── En-tête / pied de page ──────────────────────────────────
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\fancyhead[R]{\footnotesize\sffamily\rightmark}
\fancyfoot[C]{}

% ── Styles de titres ────────────────────────────────────────
\usepackage{titlesec}

% # Section = Titre du document  →  supprimé (ne pas afficher)
\newcommand{\cifasSuppressTitle}[1]{}
\titleformat{\section}[block]
  {}{}
  {0pt}{\cifasSuppressTitle}
\titlespacing*{\section}{0pt}{0pt}{0pt}

% ## Subsection = Jour / Bloc  →  boîte dans la marge + séparateur
\newcommand{\cifasSubsecBloc}[1]{%
  \cifasBlocNote{#1}%
  \vspace{6pt}%
  {\color{cifasGris}\hrule height 0.4pt}%
  \vspace{6pt}%
}
\titleformat{\subsection}[block]
  {}{}
  {0pt}{\cifasSubsecBloc}[\vspace{0pt}]
\titlespacing{\subsection}{0pt}{22pt}{10pt}

% #### Subsubsection = COMMUNICATIONS LIBRES / SYMPOSIUMS  →  caps gras taille normale
\newcommand{\cifasSubsubHead}[1]{%
  {\sffamily\large\bfseries\MakeUppercase{#1}}%
}
\titleformat{\subsubsection}[block]
  {}{}
  {0pt}{\cifasSubsubHead}
\titlespacing{\subsubsection}{0pt}{10pt}{4pt}

% ##### Paragraph = Titre de présentation  →  grand serif gras, toujours en bloc
\newcommand{\cifasPresTitle}[1]{{\normalfont\Large\bfseries #1}\par\nobreak\vspace{3pt}}
\titleformat{\paragraph}[block]
  {}{}
  {0pt}{\cifasPresTitle}
\titlespacing*{\paragraph}{0pt}{10pt}{0pt}

% ###### Subparagraph = sous-titre discret
\titleformat{\subparagraph}[block]
  {\rmfamily\small\itshape}{}
  {0pt}{}
\titlespacing{\subparagraph}{0pt}{2pt}{2pt}

% ── Soulignement sécable (filtre Lua : \uline) ──────────────
